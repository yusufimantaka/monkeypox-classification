% ----- SECTION: INTRODUCTION -----
\section{Introduction}

\begin{frame}
    \frametitle{Background}
    
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \textbf{Monkeypox (MPX)} is a zoonotic disease caused by the Monkeypox virus.
            \begin{itemize}
                \item Historically endemic to Central and West Africa
                \item 2022--2023 global outbreak highlighted urgent need for rapid diagnostics
                \item Visual inspection of skin lesions is the primary diagnostic method
                \item Easily confused with chickenpox, measles, and other skin rashes
            \end{itemize}
            \vspace{0.5em}
            \alert{Challenge:} Automated classification is needed for timely diagnosis and containment.
        \end{column}
        
        \begin{column}{0.4\textwidth}
            \includegraphics[width=\textwidth]{sample_images.png}
            \centerline{\small Sample skin lesion images}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}
    \frametitle{Problem Statement \& Objectives}
    
    \begin{block}{Problem Statement}
        Skin lesions from Monkeypox are visually similar to other conditions, making manual diagnosis error-prone and time-consuming.
    \end{block}
    
    \vspace{0.5em}
    
    \begin{exampleblock}{Research Objectives}
        \begin{enumerate}
            \item Implement robust lesion segmentation from skin background
            \item Extract rich handcrafted features (color, texture, shape, edges, frequency)
            \item Leverage transfer-learned CNN embeddings (ResNet18)
            \item Compare handcrafted, CNN, and hybrid feature representations
            \item Evaluate four classifiers for binary MPX vs. non-MPX classification
        \end{enumerate}
    \end{exampleblock}
\end{frame}

\section{Dataset \& Preprocessing}

\begin{frame}
    \frametitle{Dataset Description}
    
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \textbf{Source:} Monkeypox Skin Lesion Dataset (Kaggle)
            \begin{itemize}
                \item 228 skin lesion images (224$\times$224 RGB)
                \item \textbf{Monkeypox (MPX):} 102 images
                \item \textbf{Non-Monkeypox (ETC):} 126 images
                \item Slightly imbalanced: $\sim$45\% MPX, 55\% ETC
            \end{itemize}
            \vspace{0.5em}
            \textbf{Split:} Stratified 70/30
            \begin{itemize}
                \item Training: 160 images
                \item Testing: 68 images
            \end{itemize}
        \end{column}
        
        \begin{column}{0.45\textwidth}
            \includegraphics[width=\textwidth]{sample_images.png}
            \centerline{\small Dataset sample images}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}
    \frametitle{Lesion Segmentation Pipeline}
    
    \begin{columns}[T]
        \begin{column}{0.45\textwidth}
            \textbf{Method:} LAB + CLAHE + Otsu + Morphology
            \vspace{0.3em}
            \begin{enumerate}
                \small
                \item Convert BGR $\rightarrow$ LAB color space
                \item Apply CLAHE to $L^*$ and $a^*$ channels
                \item Combine $L^*$ and inverted $a^*$ (weights 0.3, 0.7)
                \item Otsu thresholding for binarization
                \item Morphological closing + opening ($5\times5$ ellipse)
                \item Keep largest connected component
            \end{enumerate}
            \vspace{0.3em}
            \alert{Result:} Clean binary mask isolating the primary lesion.
        \end{column}
        
        \begin{column}{0.5\textwidth}
            \includegraphics[width=\textwidth]{segmentation_example.png}
            \centerline{\small Segmentation results}
        \end{column}
    \end{columns}
\end{frame}

% ----- SECTION: METHODOLOGY -----
\section{Methodology}

\begin{frame}
    \frametitle{Handcrafted Feature Extraction}
    
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Seven categories, $\sim$200 raw features}
            \vspace{0.3em}
            \begin{itemize}
                \small
                \item \textbf{Color:} Moments + HSV histograms + percentiles in BGR/HSV/LAB (84)
                \item \textbf{GLCM:} Contrast, dissimilarity, homogeneity, energy, correlation (10)
                \item \textbf{LBP:} Uniform patterns, radius=2, P=16 (18)
                \item \textbf{Gabor:} 3 orientations $\times$ 2 scales, mean+std (12)
                \item \textbf{Shape:} Hu moments + geometric properties (12)
                \item \textbf{Edge:} Canny density + Sobel gradient stats (3)
                \item \textbf{FFT:} Radial power spectrum rings (16)
            \end{itemize}
            \vspace{0.3em}
            After variance filtering: \alert{155 features}
        \end{column}
        
        \begin{column}{0.48\textwidth}
            \textbf{Feature importance (by category)}
            \vspace{0.3em}
            \includegraphics[width=\textwidth]{category_importance.png}
            \centerline{\small Cumulative Gini importance}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}
    \frametitle{CNN Feature Extraction}
    
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \textbf{Transfer Learning with ResNet18}
            \vspace{0.3em}
            \begin{itemize}
                \item Pre-trained on ImageNet (1.2M images, 1000 classes)
                \item Remove final FC layer
                \item Extract 512-dim embedding from global average pooling
                \item Inference mode (no gradient computation)
            \end{itemize}
            \vspace{0.5em}
            \textbf{Preprocessing}
            \begin{itemize}
                \item Resize to 224$\times$224
                \item ImageNet normalization
                \item $\mu = [0.485, 0.456, 0.406]$
                \item $\sigma = [0.229, 0.224, 0.225]$
            \end{itemize}
        \end{column}
        
        \begin{column}{0.45\textwidth}
            \begin{tcolorbox}[colback=ugmLightGrey,colframe=ugmBlue,title=Feature Representations]
                \begin{enumerate}
                    \small
                    \item \textbf{Handcrafted only} (155 dims)
                    \item \textbf{CNN only} (512 dims)
                    \item \textbf{Hybrid (full)} (667 dims)
                    \item \textbf{Hybrid (selected)} (200 dims via MI)
                \end{enumerate}
            \end{tcolorbox}
            All features Z-score normalized before classification.
        \end{column}
    \end{columns}
\end{frame}

\section{Results}

\begin{frame}
    \frametitle{Classification Results}
    
    \begin{table}
        \centering
        \scriptsize
        \begin{tabular}{@{}p{2.3cm}p{2.2cm}cccc@{}}
            \toprule
            \textbf{Feature Set} & \textbf{Classifier} & \textbf{Acc.} & \textbf{F1} & \textbf{Test AUC} & \textbf{CV AUC} \\
            \midrule
            CNN (ResNet18) & SVM (RBF) & \textbf{0.855} & \textbf{0.853} & \textbf{0.949} & 0.891 \\
            Hybrid (full) & XGBoost & 0.841 & 0.838 & 0.890 & 0.798 \\
            Hybrid (selected) & XGBoost & 0.855 & 0.848 & 0.907 & 0.833 \\
            Hybrid (selected) & SVM (RBF) & 0.841 & 0.840 & 0.923 & 0.916 \\
            CNN (ResNet18) & Random Forest & 0.812 & 0.806 & 0.911 & 0.821 \\
            CNN (ResNet18) & Logistic Reg. & 0.826 & 0.823 & 0.887 & 0.847 \\
            Hybrid (full) & SVM (RBF) & 0.826 & 0.825 & 0.928 & 0.915 \\
            Handcrafted & XGBoost & 0.754 & 0.752 & 0.840 & 0.768 \\
            Handcrafted & SVM (RBF) & 0.594 & 0.592 & 0.711 & 0.776 \\
            \bottomrule
        \end{tabular}
    \end{table}

% ----- SECTION: DISCUSSION -----
\section{Discussion}

\begin{frame}
    \frametitle{Discussion}
    
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{block}{CNN Features}
                \begin{itemize}
                    \item ResNet18 embeddings highly effective even \textbf{without fine-tuning}
                    \item Capture high-level patterns: shape, color, texture
                    \item Best performance: AUC = 0.949
                \end{itemize}
            \end{block}
            
            \begin{exampleblock}{Handcrafted Features}
                \begin{itemize}
                    \item Moderate performance (best AUC = 0.840)
                    \item Color and GLCM most discriminative
                    \item Interpretable and lightweight
                \end{itemize}
            \end{exampleblock}
        \end{column}
        
        
        \begin{column}{0.48\textwidth}
            \begin{alertblock}{Hybrid Approach}
                \begin{itemize}
                    \item Did \textbf{not} surpass CNN-only baseline
                    \item Small dataset (228 images) causes overfitting with 667 dims
                    \item Feature selection (200 dims) helps but still below CNN-only
                    \item May outperform with larger dataset
                \end{itemize}
            \end{alertblock}
            
            \begin{tcolorbox}[colback=ugmLightGrey,colframe=ugmBlue,title=Classifier Choice]
                SVM (RBF) consistently best across all feature sets due to ability to model non-linear decision boundaries in high-dimensional spaces.
            \end{tcolorbox}
        \end{column}
    \end{columns}
\end{frame}
